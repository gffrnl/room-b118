\section{Caso linear}
Vamos trabalhar os cancelamentos na seguinte expressão, que apresenta cancelamento catastrófico quando $k$ é grande.
\begin{eqnarray*}
    G'(k+1) - 2G'(k) + G'(k-1) &=&\frac{1}{\alpha(\alpha-1)}\left[|k+1|^{1-\alpha}-2|k|^{1-\alpha} + |k-1|^{1-\alpha}\right] \\
    &=&\frac{|k|^{1-\alpha}}{\alpha(\alpha-1)}\left[|1+1/k|^{1-\alpha}-2 + |1-1/k|^{1-\alpha}\right]\\
    &=&\frac{|k|^{1-\alpha}}{\alpha(\alpha-1)}\left[\left(e^{(1-\alpha)\ln(1+1/k)}-1\right) + \left(e^{(1-\alpha)\ln(1-1/k)}-1\right)\right]
\end{eqnarray*}

Além disso a série bionomial no dá a assintótica:
\begin{eqnarray*}
    (1+x)^{1-\alpha} + (1-x)^{1-\alpha} &=& 2 + (\alpha - 1) \alpha x^2 + \frac{(\alpha - 1) \alpha (\alpha + 1) (\alpha + 2)}{4!/2} x^4\\ &+& \frac{(\alpha - 1) \alpha (\alpha + 1) (\alpha + 2) (\alpha + 3) (\alpha + 4)}{6!/2} x^6 +\cdots
\end{eqnarray*}

Definindo $x=1/k$, temos:

\begin{eqnarray*}
    G'(k+1) - 2G'(k) + G'(k-1) &\sim& -|k|^{-\alpha-1}\left[1 + \frac{(1+\alpha)(2+\alpha)}{12}|k|^{-2}+ \cdots \right]
\end{eqnarray*}

$$E = |k|^{-\alpha-1}\sum_{k=2}^\infty a_n (\alpha+ 3)\cdots (\alpha + n -2)x^n$$ 
onde $a_n = 0$ se n é ímpar e $a_n = \frac{2}{n!}$ se n é par.

\section{Quadrático - Termos pares e alpha != 1}
\begin{eqnarray*}
E&:=&  G'(k+1) + G'(k-1) - G(k+1) + G(k-1)\\
&=& \frac{1}{\alpha(\alpha-1)}\left[|k+1|^{1-\alpha} + |k-1|^{1-\alpha}\right] 
+ \frac{1}{\alpha(\alpha-1)(2-\alpha)}\left[-(k+1)|k+1|^{1-\alpha} + (k-1)|k-1|^{1-\alpha}\right]\\
&=& \frac{1}{\alpha(\alpha-1)}\left[E_1 + 
 \frac{E_2}{(2-\alpha)}\right]
\end{eqnarray*}
\begin{eqnarray*}
    E_1&:=&  |k+1|^{1-\alpha} + |k-1|^{1-\alpha}\\
&=&k^{1-\alpha}\left[(1+1/k)^{1-\alpha} + (1-1/k)^{1-\alpha}\right]
\end{eqnarray*}


\begin{eqnarray*}
    E_2&:=&  \left[-(k+1)^{2-\alpha} + (k-1)^{2-\alpha}\right]\\
    &=&-k^{2-\alpha}\left[(1+1/k)^{2-\alpha}-(1-1/k)^{2-\alpha}\right]
\end{eqnarray*}


Portanto:
\begin{eqnarray*}
    E_2
    &=&-|k|^{2-\alpha}\left[(1+1/k)^{2-\alpha}-(1-1/k)^{2-\alpha}\right]
\end{eqnarray*}


Para obter a assintótica de $E_1$ e $E_2$, vamos usar a série binomial com $x = \frac{1}{k}$:

\begin{eqnarray*}
    (1+x)^{1-\alpha} + (1-x)^{1-\alpha} &=& 2 + (\alpha - 1) \alpha x^2 + \frac{(\alpha - 1) \alpha (\alpha + 1) (\alpha + 2)}{4!/2} x^4\\ &+& \frac{(\alpha - 1) \alpha (\alpha + 1) (\alpha + 2) (\alpha + 3) (\alpha + 4)}{6!/2} x^6 +\cdots
\end{eqnarray*}
\begin{eqnarray*}
    \frac{(1+x)^{2-\alpha} -(1-x)^{2-\alpha}}{x(2-\alpha)} = 2 +\frac{(\alpha-1)\alpha}{3!/2}x^2 +\frac{(\alpha-1)\alpha(\alpha+1)(\alpha+2)}{5!/2}x^4 + \cdots
\end{eqnarray*}

portanto:
\begin{eqnarray*}
    (1+x)^{1-\alpha} + (1-x)^{1-\alpha} - \frac{(1+x)^{2-\alpha} - (1-x)^{2-\alpha}}{x(2-\alpha)} = \sum_{n=2}^\infty a_n(\alpha-1)\alpha\cdots (\alpha+n-2)x^n
\end{eqnarray*}
onde  $a_n=0$ se $n$ é ímpar e
$a_n = \frac{2}{n!}-\frac{2}{(n+1)!}=\frac{2n}{(n+1)!}$
E finalmente:
\begin{eqnarray*}
    E&=& \sum_{n=2}^\infty a_n(\alpha+1)\cdots (\alpha+n-2)x^n
\end{eqnarray*}

\begin{eqnarray*}
    E &=& \frac{1}{2-\alpha} - G''(1) - \frac{G'(3)+3G'(1)}{2} + G(3)-G(1)\\
    &=&\frac{1}{2-\alpha} + \frac{1}{\alpha} - \frac{3^{1-\alpha} + 3}{2\alpha(\alpha-1)}+\frac{3^{2-\alpha} - 1}{\alpha(\alpha-1)(2-\alpha)}\\
    &=&\frac{2}{\alpha(2-\alpha)} + \frac{(3^{1-\alpha} + 3)(\alpha-2)}{2\alpha(\alpha-1)(2-\alpha)}+\frac{6\cdot 3^{1-\alpha} - 2}{2\alpha(\alpha-1)(2-\alpha)}\\
    &=&\frac{2}{\alpha(2-\alpha)} + \frac{3^{1-\alpha}(\alpha+4)+3\alpha-8 }{2\alpha(\alpha-1)(2-\alpha)}\\
    &=&\frac{2}{\alpha(2-\alpha)} + \frac{(3^{1-\alpha}-1)(\alpha+4)+4(\alpha-1) }{2\alpha(\alpha-1)(2-\alpha)}\\
    &=&\frac{4}{\alpha(2-\alpha)} + \frac{(\alpha+4)}{2\alpha(2-\alpha)}\cdot \frac{3^{1-\alpha}-1}{\alpha -1 }
\end{eqnarray*}

$$\lim_{\alpha \to 1} \frac{3^{1-\alpha}}{\alpha-1}=-\ln(3)$$
Assim
$$\lim_{\alpha \to 1} E= 4 - \frac{5}{2}\ln(3)= \frac{8-5\ln(3)}{2}$$

\section{Integral numérica}
\begin{eqnarray*}
    I&:=&\int_0^\pi x^{\alpha}\cos(kx)dx =\sum_{n=0}^{k-1}\int_{n\theta}^{(n+1)\theta} x^{\alpha}\cos(kx)dx\\
      &=&\sum_{n=0}^{k-1}\int_{0}^{\theta} (x+n\theta)^{\alpha}\cos(kx + n\pi)dx\\
      &=&\sum_{n=0}^{k-1}(-1)^k\int_{0}^{\theta} (x+n\theta)^{\alpha}\cos(kx)dx\\
      &=&\sum_{n=0}^{k-1}(-1)^k\left[\int_{0}^{\theta/2} (x+n\theta)^{\alpha}\cos(kx)dx+ \int_{\theta/2}^{\theta} (x+n\theta)^{\alpha}\cos(kx)dx\right]\\
\end{eqnarray*}
onde $\theta = \frac{\pi}{k}$.
Na segunda integral, faça a mudança de variável $y= \theta - x$:
\begin{eqnarray*}
    \int_{\theta/2}^{0} (-y+(n+1)\theta)^{\alpha}\cos(-ky+\pi)(-dy) &=& -\int_0^{\theta/2} (-y+(n+1)\theta)^{\alpha}\cos(ky)dy
\end{eqnarray*}
Portanto:
\begin{eqnarray*}
    I&:=&\sum_{n=0}^{k-1}(-1)^k\int_{0}^{\theta/2}\left[ (x+n\theta)^{\alpha} - (-x + (n+1)\theta)^{\alpha}\right]\cos(kx)dx\\
    &:=& \sum_{n=0}^{k-1}(-1)^k\int_{0}^{\theta/2}\left[ (x+n\theta)^{\alpha} - (-x + (n+1)\theta)^{\alpha}\right]\cos(kx)dx
\end{eqnarray*}
