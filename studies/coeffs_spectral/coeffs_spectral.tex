\section{Série}
\begin{equation}
  {}_1F_2\left(\frac{\alpha+1}{2};\frac{1}{2},\frac{\alpha+3}{2};-\frac{(k\pi)^2}{4}\right)
  =\sum_{n=0}^{\infty} \frac{(a_1)_n}{(b_1)_n(b_2)_n}
   \frac{z^n}{n!}
  \;.
\end{equation}
\begin{equation}
\begin{split}
(a)_0 &= 1 \\
(a)_n &= a\left(a+1\right)\left(a+2\right)\ldots\left(a+n-1\right)\,,\quad n\geq 1
\end{split}
\end{equation}

\begin{equation}
  {}_1F_2\left(\frac{\alpha+1}{2};\frac{1}{2},\frac{\alpha+3}{2};-\frac{(k\pi)^2}{4}\right)
  =1 + \sum_{n=1}^{\infty} \frac{(a_1)_n}{(b_1)_n(b_2)_n}
   \frac{z^n}{n!}
  \;.
\end{equation}

Para $n \geq 1$
\begin{equation}
\begin{split}
(a_1)_n &= a_1\left(a_1+1\right)\left(a_1+2\right)\ldots\left(a_1+n-1\right) \\
(a_1)_n &= \frac{\alpha+1}{2}\left(\frac{\alpha+1}{2}+1\right)\left(\frac{\alpha+1}{2}+2\right)\ldots\left(\frac{\alpha+1}{2}+n-1\right) \\
(a_1)_n &= \frac{1}{2^n}\left(\alpha+1\right)\left(\alpha+3\right)\left(\alpha+5\right)\ldots\left(\alpha+2n-1\right) \\
\end{split}
\end{equation}

\begin{equation}
\begin{split}
(b_1)_n &= b_1\left(b_1+1\right)\left(b_1+2\right)\ldots\left(b_1+n-1\right) \\
(b_1)_n &= \frac{1}{2}\left(\frac{1}{2}+1\right)\left(\frac{1}{2}+2\right)\ldots\left(\frac{1}{2}+n-1\right) \\
(b_1)_n &= \frac{1}{2^n} 3\cdot 5\cdot\ldots\cdot \left(2n-1\right) \\
\end{split}
\end{equation}

\begin{equation}
\begin{split}
(b_2)_n &= b_2\left(b_2+1\right)\left(b_2+2\right)\ldots\left(b_2+n-1\right) \\
(b_2)_n &= \frac{\alpha+3}{2}\left(\frac{\alpha+3}{2}+1\right)\left(\frac{\alpha+3}{2}+2\right)\ldots\left(\frac{\alpha+3}{2}+n-1\right) \\
(b_2)_n &= \frac{1}{2^n}\left(\alpha+3\right)\left(\alpha+5\right)\left(\alpha+7\right)\ldots\left(\alpha+2n+1\right) \\
\end{split}
\end{equation}

\begin{equation}
\begin{split}
\frac{(a_1)_n}{(b_2)_n} = \frac{\left(\alpha+1\right)\left(\alpha+3\right)\left(\alpha+5\right)\ldots\left(\alpha+2n-1\right)}{\left(\alpha+3\right)\left(\alpha+5\right)\left(\alpha+7\right)\ldots\left(\alpha+2n+1\right)}
&= \frac{(\alpha+1)}{(\alpha+2n+1)} \\
&= \frac{\alpha+1}{\alpha+1+2n} 
\end{split}
\end{equation}

\begin{equation}
\sum_{n=1}^{\infty} \frac{(a_1)_n}{(b_1)_n(b_2)_n}
   \frac{z^n}{n!}
   = \frac{\alpha+1}{\alpha+1+2n} \frac{2^n}{\left(3\cdot 5\cdot\ldots\cdot \left(2n-1\right)\right)n!} z^n
  \;.
\end{equation}

\begin{equation}
\frac{(a_1)_n}{(b_1)_n(b_2)_n}
   \frac{z^n}{n!}
   = \frac{\alpha+1}{\alpha+2n+1} \frac{2^n}{\left(3\cdot 5\cdot\ldots\cdot \left(2n-1\right)\right)n!} z^n
  \;.
\end{equation}

\begin{equation}
\begin{split}
\frac{(a_1)_{n+1}}{(b_1)_{n+1}(b_2)_{n+1}}
   \frac{z^{n+1}}{(n+1)!}
   &= \frac{\alpha+1}{\alpha+1+2(n+1)} \frac{2^{n+1}}{\left(3\cdot 5\cdot\ldots\cdot \left(2(n+1)-1\right)\right)(n+1)!} z^{n+1} \\
   &= \frac{\alpha+1}{\alpha+2n+2} \frac{2^{n+1}}{\left(3\cdot 5\cdot\ldots\cdot \left(2n+1\right)\right)(n+1)n!} z^{n+1}
  \;.
\end{split}
\end{equation}

\begin{equation}
\begin{split}
\frac{(a_1)_{n+1}}{(b_1)_{n+1}(b_2)_{n+1}}
   \frac{z^{n+1}}{(n+1)!}
   &= \frac{\alpha+1}{\alpha+2n+2} \frac{2^{n+1}}{\left(3\cdot 5\cdot\ldots\cdot \left(2n+1\right)\right)(n+1)n!} z^{n+1}
  \;.
\end{split}
\end{equation}

Chamando
\begin{equation}
C_n = \frac{(a_1)_n}{(b_1)_n(b_2)_n} \frac{z^n}{n!}
\end{equation}
temos
\begin{equation}
\frac{C_{n+1}}{C_n}=
\frac{\frac{\alpha+1}{\alpha+2n+2} \frac{2^{n+1}}{\left(3\cdot 5\cdot\ldots\cdot \left(2n+1\right)\right)(n+1)n!} z^{n+1}}{\frac{\alpha+1}{\alpha+2n+1} \frac{2^n}{\left(3\cdot 5\cdot\ldots\cdot \left(2n-1\right)\right)n!} z^n}
=2\frac{\alpha+2n+1}{\alpha+2n+2}\frac{z}{(n+1)(2n+1)}
\end{equation}

\begin{equation}
\frac{\alpha+2n+1}{\alpha+2n+2} = 1 - \frac{1}{\alpha+2n+2} = 1 - \frac{1}{\alpha+2(n+1)}
\end{equation}

\begin{equation}
\frac{2}{(n+1)(2n+1)} = \frac{2}{2n^2+n+2n+2} = \frac{2}{2n^2+3n+2} = \frac{1}{n^2+3n/2+1}
\end{equation}

Portanto,
\begin{equation}
\begin{split}
C_1 &= \frac{(a_1)_1}{(b_1)_1(b_2)_1} z \\
C_{n+1} &=  C_n \left(1 - \frac{1}{\alpha+2(n+1)}\right)
\left(\frac{z}{n^2+3n/2+1}\right) \\
\end{split}
\end{equation}

\begin{equation}
\begin{split}

\end{split}
\end{equation}
